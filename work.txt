queue = []

def printTree(rootNode, queue):
    if rootNode.data is not None:
        print(rootNode.data)

    if len(queue) > 0:
        t = queue[len(queue)-1]
        queue.remove([len(queue)-1])
        printTree(t, queue)

    if rootNode is not None and hasattr(rootNode, "left"):
        queue.append(rootNode.right)
        printTree(rootNode.left, queue)





printTree(root)
