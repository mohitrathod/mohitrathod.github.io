 def removeAdjoints(ipx, start, end):
    print(ipx)
    if len(ipx) > 0 and end < len(ipx):
        if ipx[start] == ipx[end]:
            if end + 1 < len(ipx):
                removeAdjoints(ipx, start, end + 1)
            elif start < end:
                ipx = ipx[0:start]
                removeAdjoints(ipx, 0, 1)
        elif start + 1 < end:
            ipx = ipx[0:start] + ipx[end:len(ipx)]
            removeAdjoints(ipx, 0, 1)
        else:
            if end + 1 < len(ipx):
                removeAdjoints(ipx, start + 1, end + 1)
